#include <iostream>

using namespace std;

int main() {
    int Num;
    cout << "Enter a number: ";
    cin >> Num;

    int a = Num * Num;
    int b = Num * Num * Num;
    int c = Num * Num * Num * Num;

    cout << a << endl;
    cout << b << endl;
    cout << c << endl;

    return 0;
}
